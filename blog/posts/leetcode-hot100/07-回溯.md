# 07 - 回溯（10 题）

回溯题的通用套路很稳定：**路径、选择列表、结束条件、撤销选择**。难点主要在剪枝。

## 61. 全排列

- **标签**：回溯

### 思路

路径里放当前已经选过的数，用 `used` 数组防止重复选择。

### 代码（JavaScript）

```js
function permute(nums) {
  const res = [];
  const path = [];
  const used = new Array(nums.length).fill(false);
  function dfs() {
    if (path.length === nums.length) {
      res.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue;
      used[i] = true;
      path.push(nums[i]);
      dfs();
      path.pop();
      used[i] = false;
    }
  }
  dfs();
  return res;
}
```

## 62. 子集

- **标签**：回溯

### 思路

每个元素都面临“选 / 不选”，所以递归时先收集当前路径。

### 代码（JavaScript）

```js
function subsets(nums) {
  const res = [];
  const path = [];
  function dfs(start) {
    res.push([...path]);
    for (let i = start; i < nums.length; i++) {
      path.push(nums[i]);
      dfs(i + 1);
      path.pop();
    }
  }
  dfs(0);
  return res;
}
```

## 63. 电话号码的字母组合

- **标签**：回溯 / 字符串

### 思路

每个数字映射多个字母，逐层枚举当前位置可填的字符。

### 代码（JavaScript）

```js
function letterCombinations(digits) {
  if (!digits.length) return [];
  const map = {
    2: 'abc', 3: 'def', 4: 'ghi', 5: 'jkl',
    6: 'mno', 7: 'pqrs', 8: 'tuv', 9: 'wxyz'
  };
  const res = [];
  function dfs(index, path) {
    if (index === digits.length) {
      res.push(path);
      return;
    }
    for (const ch of map[digits[index]]) dfs(index + 1, path + ch);
  }
  dfs(0, '');
  return res;
}
```

## 64. 组合总和

- **标签**：回溯

### 思路

因为元素可重复使用，所以递归进入下一层时仍从当前下标开始。

### 代码（JavaScript）

```js
function combinationSum(candidates, target) {
  const res = [];
  const path = [];
  function dfs(start, remain) {
    if (remain === 0) {
      res.push([...path]);
      return;
    }
    if (remain < 0) return;
    for (let i = start; i < candidates.length; i++) {
      path.push(candidates[i]);
      dfs(i, remain - candidates[i]);
      path.pop();
    }
  }
  dfs(0, target);
  return res;
}
```

## 65. 括号生成

- **标签**：回溯

### 思路

左括号数量没用完就能放左括号，右括号数量小于左括号数量时才能放右括号。

### 代码（JavaScript）

```js
function generateParenthesis(n) {
  const res = [];
  function dfs(path, left, right) {
    if (path.length === 2 * n) {
      res.push(path);
      return;
    }
    if (left < n) dfs(path + '(', left + 1, right);
    if (right < left) dfs(path + ')', left, right + 1);
  }
  dfs('', 0, 0);
  return res;
}
```

## 66. 分割回文串

- **标签**：回溯 / 字符串

### 思路

枚举当前位置的所有前缀，只要是回文串就可以进入下一层递归。

### 代码（JavaScript）

```js
function partition(s) {
  const res = [];
  const path = [];
  function isPal(str, l, r) {
    while (l < r) if (str[l++] !== str[r--]) return false;
    return true;
  }
  function dfs(start) {
    if (start === s.length) {
      res.push([...path]);
      return;
    }
    for (let end = start; end < s.length; end++) {
      if (!isPal(s, start, end)) continue;
      path.push(s.slice(start, end + 1));
      dfs(end + 1);
      path.pop();
    }
  }
  dfs(0);
  return res;
}
```

## 67. N 皇后

- **标签**：回溯 / 棋盘

### 思路

逐行放皇后，列、主对角线、副对角线都不能冲突。

### 代码（JavaScript）

```js
function solveNQueens(n) {
  const res = [];
  const board = Array.from({ length: n }, () => Array(n).fill('.'));
  const cols = new Set(), diag1 = new Set(), diag2 = new Set();
  function dfs(row) {
    if (row === n) {
      res.push(board.map(item => item.join('')));
      return;
    }
    for (let col = 0; col < n; col++) {
      if (cols.has(col) || diag1.has(row - col) || diag2.has(row + col)) continue;
      cols.add(col); diag1.add(row - col); diag2.add(row + col);
      board[row][col] = 'Q';
      dfs(row + 1);
      board[row][col] = '.';
      cols.delete(col); diag1.delete(row - col); diag2.delete(row + col);
    }
  }
  dfs(0);
  return res;
}
```

## 68. 复原 IP 地址

- **标签**：回溯

### 思路

每段长度只能是 1~3，且要满足数值合法、不能有前导 0。

### 代码（JavaScript）

```js
function restoreIpAddresses(s) {
  const res = [];
  const path = [];
  function dfs(start) {
    if (path.length === 4) {
      if (start === s.length) res.push(path.join('.'));
      return;
    }
    for (let len = 1; len <= 3; len++) {
      if (start + len > s.length) break;
      const part = s.slice(start, start + len);
      if ((part.length > 1 && part[0] === '0') || Number(part) > 255) continue;
      path.push(part);
      dfs(start + len);
      path.pop();
    }
  }
  dfs(0);
  return res;
}
```

## 69. 单词搜索 II

- **标签**：Trie / 回溯

### 思路

先把所有单词建成 Trie，再在棋盘上 DFS，走不通时能尽早剪枝。

### 代码（JavaScript）

```js
function findWords(board, words) {
  const root = {};
  for (const word of words) {
    let node = root;
    for (const ch of word) node = node[ch] ||= {};
    node.word = word;
  }
  const res = [];
  const m = board.length, n = board[0].length;
  function dfs(i, j, node) {
    const ch = board[i][j];
    if (!node[ch]) return;
    node = node[ch];
    if (node.word) {
      res.push(node.word);
      node.word = null;
    }
    board[i][j] = '#';
    for (const [x, y] of [[i+1,j],[i-1,j],[i,j+1],[i,j-1]]) {
      if (x>=0 && x<m && y>=0 && y<n && board[x][y] !== '#') dfs(x, y, node);
    }
    board[i][j] = ch;
  }
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) dfs(i, j, root);
  }
  return res;
}
```

## 70. 组合总和 II

- **标签**：回溯 / 去重

### 思路

元素只能用一次，所以递归进下一层要从 `i + 1` 开始，并跳过同层重复值。

### 代码（JavaScript）

```js
function combinationSum2(candidates, target) {
  candidates.sort((a, b) => a - b);
  const res = [];
  const path = [];
  function dfs(start, remain) {
    if (remain === 0) {
      res.push([...path]);
      return;
    }
    for (let i = start; i < candidates.length; i++) {
      if (i > start && candidates[i] === candidates[i - 1]) continue;
      if (candidates[i] > remain) break;
      path.push(candidates[i]);
      dfs(i + 1, remain - candidates[i]);
      path.pop();
    }
  }
  dfs(0, target);
  return res;
}
```
