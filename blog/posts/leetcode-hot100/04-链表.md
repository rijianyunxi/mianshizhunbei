# 04 - 链表（10 题）

写链表题时，一定要养成两个习惯：**先画指针变化，再动代码；能用哑节点时就用哑节点。**

## 31. 反转链表

- **标签**：链表

### 思路

用 `prev / cur / next` 三指针逐个反转指向。

### 代码（JavaScript）

```js
function reverseList(head) {
  let prev = null, cur = head;
  while (cur) {
    const next = cur.next;
    cur.next = prev;
    prev = cur;
    cur = next;
  }
  return prev;
}
```

## 32. 回文链表

- **标签**：链表 / 双指针

### 思路

快慢指针找中点，反转后半段，再从两头同时比较。

### 代码（JavaScript）

```js
function isPalindrome(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  let prev = null;
  while (slow) {
    const next = slow.next;
    slow.next = prev;
    prev = slow;
    slow = next;
  }
  while (prev) {
    if (head.val !== prev.val) return false;
    head = head.next;
    prev = prev.next;
  }
  return true;
}
```

## 33. 环形链表

- **标签**：快慢指针

### 思路

快指针一次走两步，慢指针一次走一步，如果能相遇就有环。

### 代码（JavaScript）

```js
function hasCycle(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) return true;
  }
  return false;
}
```

## 34. 环形链表 II

- **标签**：快慢指针

### 思路

相遇后让一个指针回到头节点，两个指针再同步前进，相遇点就是入环点。

### 代码（JavaScript）

```js
function detectCycle(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) {
      let p = head;
      while (p !== slow) {
        p = p.next;
        slow = slow.next;
      }
      return p;
    }
  }
  return null;
}
```

## 35. 相交链表

- **标签**：链表

### 思路

两个指针分别走 A+B 和 B+A，走完后长度被自动对齐。

### 代码（JavaScript）

```js
function getIntersectionNode(headA, headB) {
  let p1 = headA, p2 = headB;
  while (p1 !== p2) {
    p1 = p1 ? p1.next : headB;
    p2 = p2 ? p2.next : headA;
  }
  return p1;
}
```

## 36. 合并两个有序链表

- **标签**：链表

### 思路

用哑节点把两个有序链表按大小顺序串起来。

### 代码（JavaScript）

```js
function mergeTwoLists(list1, list2) {
  const dummy = new ListNode(0);
  let cur = dummy;
  while (list1 && list2) {
    if (list1.val < list2.val) {
      cur.next = list1;
      list1 = list1.next;
    } else {
      cur.next = list2;
      list2 = list2.next;
    }
    cur = cur.next;
  }
  cur.next = list1 || list2;
  return dummy.next;
}
```

## 37. 两两交换链表中的节点

- **标签**：链表

### 思路

每次取两个节点做一次局部交换，哑节点方便处理头部变化。

### 代码（JavaScript）

```js
function swapPairs(head) {
  const dummy = new ListNode(0, head);
  let cur = dummy;
  while (cur.next && cur.next.next) {
    const first = cur.next;
    const second = cur.next.next;
    first.next = second.next;
    second.next = first;
    cur.next = second;
    cur = first;
  }
  return dummy.next;
}
```

## 38. K 个一组翻转链表

- **标签**：链表

### 思路

每次截出 k 个节点翻转，不够 k 个就保持原样。

### 代码（JavaScript）

```js
function reverseKGroup(head, k) {
  const dummy = new ListNode(0, head);
  let pre = dummy;
  while (true) {
    let tail = pre;
    for (let i = 0; i < k && tail; i++) tail = tail.next;
    if (!tail) break;
    const next = tail.next;
    let prev = next, cur = pre.next;
    while (cur !== next) {
      const tmp = cur.next;
      cur.next = prev;
      prev = cur;
      cur = tmp;
    }
    const start = pre.next;
    pre.next = tail;
    pre = start;
  }
  return dummy.next;
}
```

## 39. 删除链表的倒数第 N 个结点

- **标签**：链表 / 双指针

### 思路

快指针先走 N 步，再和慢指针同步，慢指针停在待删节点前一个位置。

### 代码（JavaScript）

```js
function removeNthFromEnd(head, n) {
  const dummy = new ListNode(0, head);
  let fast = dummy, slow = dummy;
  for (let i = 0; i < n; i++) fast = fast.next;
  while (fast.next) {
    fast = fast.next;
    slow = slow.next;
  }
  slow.next = slow.next.next;
  return dummy.next;
}
```

## 40. 排序链表

- **标签**：链表 / 归并排序

### 思路

链表适合用归并排序：先找中点拆分，再合并两个有序链表。

### 代码（JavaScript）

```js
function sortList(head) {
  if (!head || !head.next) return head;
  let slow = head, fast = head.next;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  const rightHead = slow.next;
  slow.next = null;
  return merge(sortList(head), sortList(rightHead));
}
function merge(a, b) {
  const dummy = new ListNode(0);
  let cur = dummy;
  while (a && b) {
    if (a.val < b.val) {
      cur.next = a;
      a = a.next;
    } else {
      cur.next = b;
      b = b.next;
    }
    cur = cur.next;
  }
  cur.next = a || b;
  return dummy.next;
}
```
